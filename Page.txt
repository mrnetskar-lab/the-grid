/* Tone modulations: hue shift & lightness for bubbles/puddle */
const TONE_MOD = {
  soft:     { hShift:   8, lShift:  6, glow: 0.10 },
  tender:   { hShift:  14, lShift: 10, glow: 0.14 },

  cold:     { hShift: -20, lShift:-12, glow: 0.06 },

};

function setChannelHue(h) {
  document.documentElement.style.setProperty("--channel-hue", h);
}

/* ── Render sidebar channel list ── */
function renderChannels() {
  const list = document.getElementById("channelList");
  list.innerHTML = "";
  Object.keys(CHARS).forEach(key => {
    const c = CHARS[key];
    const btn = document.createElement("button");
    btn.className = "channel" + (key === DATA.activeKey ? " active" : "");
    btn.style.setProperty("--ch-hue", c.hue);
    btn.onclick = () => { DATA.activeKey = key; renderAll(); };
    btn.innerHTML = `
      <span class="avatar-ring" data-ring="${c.state}" style="--ch-hue: ${c.hue};">
        <span class="avatar has-img" style="--ch-hue: ${c.hue};">
          <img src="${c.img}" alt="" onerror="this.style.display='none'" />
          <span class="initial">${c.name[0]}</span>
        </span>
      </span>
      <span class="ch-body">
        <span class="ch-name">${c.name}</span>
        <span class="ch-preview">${
          c.typing
            ? `<span class="ch-typing" style="--ch-hue:${c.hue};"><span></span><span></span><span></span></span>`
            : c.thread[c.thread.length-1].text
        }</span>
      </span>
      <span class="ch-meta">${stateLabel(c.state)}</span>
    `;
    list.appendChild(btn);
  });
}

function stateLabel(s) {
  return {dormant:"DORMANT", warming:"WARMING", pull:"PULL", opening:"OPENING", open:"OPEN", faded:"FADED"}[s] || s.toUpperCase();
}

/* ── Render chat pane ── */
function renderChat() {
  const key = DATA.activeKey;
  const c = CHARS[key];
  const tone = DATA.tone;
  const mod = TONE_MOD[tone] || TONE_MOD.neutral;

  // Global channel hue, shifted by tone
  const hue = c.hue + mod.hShift;
  setChannelHue(hue);

  // Header
  document.getElementById("focusName").textContent = c.name;
  document.getElementById("focusState").textContent = stateLabel(c.state);
  document.getElementById("focusRoute").textContent = c.route;
  document.getElementById("focusTone").textContent = tone.toUpperCase();
  document.getElementById("focusWhisper").textContent = c.whisper || "";
  document.getElementById("focusRing").setAttribute("data-ring", c.state);
  document.getElementById("focusRing").style.setProperty("--ch-hue", hue);
  document.getElementById("heroImg").src = c.img;
  // typing indicator in hero
  document.getElementById("focusTyping").style.display = c.typing ? "inline-flex" : "none";
  document.getElementById("focusIdle").style.display   = c.typing ? "none" : "inline";

  // Thread
  const t = document.getElementById("thread");
  t.innerHTML = "";
  c.thread.forEach((m, i) => {
    const div = document.createElement("div");

    const mHue = c.hue + mMod.hShift;

    div.style.setProperty("--msg-border", 0.28 + mMod.glow);
